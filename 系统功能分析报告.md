```javascript
import { Low, JSONFile } from 'lowdb';
import { Mutex } from 'async-mutex';
import fs from 'fs';
import path from 'path';
import { fileURLToPath } from 'url';

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);
const dataDir = path.join(__dirname, '../../data/db');

// 确保目录存在
if (!fs.existsSync(dataDir)) {
  fs.mkdirSync(dataDir, { recursive: true });
}

// 为每个用户创建独立的数据库
const userMutexes = new Map();

function getUserMutex(userId) {
  if (!userMutexes.has(userId)) {
    userMutexes.set(userId, new Mutex());
  }
  return userMutexes.get(userId);
}

const defaultData = {
  groups: [],
  messages: [],
  files: [],
  agents: [],
  agent_messages: [],
  memories: [],
  comments: [],
  profile: {
    nickname: '用户',
    avatar: null,
    preferences: {
      theme: 'dark',
      language: 'zh-CN'
    }
  }
};

export async function getUserDb(userId) {
  const dbFile = path.join(dataDir, `${userId}.json`);
  
  // 如果文件不存在，创建默认数据
  if (!fs.existsSync(dbFile)) {
    fs.writeFileSync(dbFile, JSON.stringify(defaultData, null, 2));
  }
  
  const adapter = new JSONFile(dbFile);
  const db = new Low(adapter, defaultData);
  
  return db;
}

// 写操作必须通过锁
export async function withWriteLock(userId, fn) {
  const mutex = getUserMutex(userId);
  return mutex.runExclusive(fn);
}

// 认证数据库
import { fileURLToPath as _f } from 'url';
const authDbFile = path.join(path.dirname(_f(import.meta.url)), '../../data/auth.json');
if (!fs.existsSync(path.dirname(authDbFile))) {
  fs.mkdirSync(path.dirname(authDbFile), { recursive: true });
}
if (!fs.existsSync(authDbFile)) {
  fs.writeFileSync(authDbFile, JSON.stringify({ sessions: [] }, null, 2));
}

export const authAdapter = new JSONFile(authDbFile);
export const authDb = new Low(authAdapter, { sessions: [] });
```

### 2.3 并发锁机制完整实现

见2.1节 `withWriteLock` 函数。核心逻辑：
- 每个用户一个独立的 `async-mutex` Mutex
- 所有写操作必须通过 `withWriteLock(userId, async () => { await db.write(); })` 执行
- 读操作不需要锁（LowDB的read是安全的）

### 3.1 SecretManager完整实现

**文件路径：`backend/src/utils/secretManager.js`**

```javascript
import crypto from 'crypto';
import dotenv from 'dotenv';
dotenv.config();

class SecretManager {
  constructor() {
    this.secrets = new Map();
    this.refreshInterval = 60 * 60 * 1000; // 1小时
    this.startAutoRefresh();
  }

  async load(key, provider = 'env') {
    let value;
    switch (provider) {
      case 'env':
        value = process.env[key];
        break;
      case 'file':
        // 文件读取逻辑
        break;
      default:
        throw new Error(`Unknown provider: ${provider}`);
    }
    
    if (value) {
      this.secrets.set(key, {
        value,
        provider,
        loadedAt: Date.now()
      });
    }
    return value;
  }

  get(key) {
    const entry = this.secrets.get(key);
    if (!entry) return null;
    return entry.value;
  }

  startAutoRefresh() {
    setInterval(async () => {
      for (const [key, entry] of this.secrets) {
        await this.load(key, entry.provider);
      }
    }, this.refreshInterval);
  }
}

export const secretManager = new SecretManager();
```

### 3.2 加密工具完整实现

**文件路径：`backend/src/utils/crypto.js`**

```javascript
import crypto from 'crypto';

export function generateToken(length = 32) {
  return crypto.randomBytes(length).toString('hex');
}

export function hashToken(token) {
  return crypto.createHash('sha256').update(token).digest('hex');
}

export function verifyToken(token, hashedToken) {
  return hashToken(token) === hashedToken;
}

export function deriveKey(password, salt, iterations = 100000, keyLength = 32) {
  return new Promise((resolve, reject) => {
    crypto.pbkdf2(password, salt, iterations, keyLength, 'sha256', (err, key) => {
      if (err) reject(err);
      resolve(key.toString('hex'));
    });
  });
}
```

### 3.3 认证中间件完整实现

**文件路径：`backend/src/middleware/auth.js`**

```javascript
import { secretManager } from '../utils/secretManager.js';

const PUBLIC_PATHS = ['/api/auth/token', '/api/auth/login', '/api/health'];

export async function authMiddleware(req, res, next) {
  // 公开路径直接放行
  if (PUBLIC_PATHS.includes(req.path)) {
    return next();
  }

  // 加载认证配置
  const authConfig = await secretManager.get('auth');
  
  // 认证未启用，放行
  if (!authConfig || authConfig.enabled === false) {
    req.user = { id: 'anonymous', role: 'user' };
    return next();
  }

  // 提取会话Cookie
  const token = req.cookies?.session_token;

  if (!token) {
    return res.status(401).json({ error: 'Unauthorized' });
  }

  // 验证会话
  const session = db.data.sessions.find(entry => entry.token === token);
  if (!session) {
    return res.status(401).json({ error: 'Unauthorized' });
  }

  req.user = { id: token, role: 'user' };
  next();
}
```

```javascript
// 消息相似度计算(基于编辑距离)
export function calculateSimilarity(str1, str2) {
  if (!str1 || !str2) return 0;
  
  const s1 = str1.toLowerCase();
  const s2 = str2.toLowerCase();
  
  if (s1 === s2) return 1;
  if (s1.length < 2 || s2.length < 2) return 0;
  
  // 简化的相似度计算(基于公共子串)
  const shorter = s1.length < s2.length ? s1 : s2;
  const longer = s1.length < s2.length ? s2 : s1;
  
  let maxMatch = 0;
  for (let i = 0; i < shorter.length - 1; i++) {
    const sub = shorter.substring(i, i + 2);
    if (longer.includes(sub)) {
      maxMatch++;
    }
  }
  
  return maxMatch / shorter.length;
}
```

```javascript
        temperature: 0.7,
        max_tokens: 2000
      })
    });

    // 处理流式响应
    const reader = response.body.getReader();
    const decoder = new TextDecoder();
    
    while (true) {
      const { done, value } = await reader.read();
      if (done) break;
      
      const chunk = decoder.decode(value, { stream: true });
      const lines = chunk.split('\n').filter(line => line.trim().startsWith('data: '));
      
      for (const line of lines) {
        const data = line.replace('data: ', '').trim();
        if (data === '[DONE]') continue;
        
        try {
          const parsed = JSON.parse(data);
          const content = parsed.choices?.[0]?.delta?.content;
          if (content) {
            fullResponse += content;
            
            // 流式推送
            if (wsServer && groupId) {

                ai: persona.id,
                content

            }
            
            if (onChunk) {
              onChunk(content);
            }
          }
        } catch {}
      }
    }
    
    if (onDone) onDone(fullResponse);
    return fullResponse;
```

```javascript
class AIModel {
  constructor(id, name) {
    this.id = id;
    this.name = name;
    this.health = true;
    this.responseTime = 0;
    this.successRate = 1.0;
    this.totalCalls = 0;
    this.failedCalls = 0;
    this.consecutiveFailures = 0;
    this.circuitBroken = false;
    this.circuitBrokenAt = null;
    this.circuitCooldown = 30000; // 30秒冷却
  }

  recordSuccess(responseTime) {
    this.totalCalls++;
    this.consecutiveFailures = 0;
    this.responseTime = responseTime;
    this.successRate = (this.totalCalls - this.failedCalls) / this.totalCalls;
    
    if (this.circuitBroken && Date.now() - this.circuitBrokenAt > this.circuitCooldown) {
      this.circuitBroken = false;
      this.circuitBrokenAt = null;
    }
  }

  recordFailure() {
    this.totalCalls++;
    this.failedCalls++;
    this.consecutiveFailures++;
    this.successRate = (this.totalCalls - this.failedCalls) / this.totalCalls;
    
    if (this.consecutiveFailures >= 3 && !this.circuitBroken) {
      this.circuitBroken = true;
      this.circuitBrokenAt = Date.now();
    }
  }

  isAvailable() {
    if (this.circuitBroken) {
      if (Date.now() - this.circuitBrokenAt > this.circuitCooldown) {
        this.circuitBroken = false;
        this.circuitBrokenAt = null;
        return true;
      }
      return false;
    }
    return this.health;
  }
}

class LoadBalancer {
  constructor() {
    this.models = new Map();
    this.weights = new Map();
  }

  registerModel(id, name) {
    this.models.set(id, new AIModel(id, name));
    this.weights.set(id, 1.0);
  }

  selectModel() {
    const available = Array.from(this.models.values()).filter(m => m.isAvailable());
    if (available.length === 0) throw new Error('No available models');
    
    // 加权选择
    let totalWeight = 0;
    for (const model of available) {
      let weight = 1.0;
      if (model.successRate >= 0.95) weight *= 1.5;
      if (model.responseTime > 5000) weight *= 0.5;
      this.weights.set(model.id, weight);
      totalWeight += weight;
    }
    
    let random = Math.random() * totalWeight;
    for (const model of available) {
      random -= this.weights.get(model.id);
      if (random <= 0) return model;
    }
    return available[available.length - 1];
  }

  getCircuitBreaker(modelId) {
    const model = this.models.get(modelId);
    return model ? {
      circuitBroken: model.circuitBroken,
      consecutiveFailures: model.consecutiveFailures,
      cooldown: model.circuitBrokenAt ? model.circuitCooldown - (Date.now() - model.circuitBrokenAt) : 0
    } : null;
  }
}
```

```javascript
function parseMentions(content) {
  const mentions = [];
  for (const [modelId, aliases] of Object.entries(AI_MENTION_ALIASES)) {
    const pattern = new RegExp(`@(${modelId}|${aliases.join('|')})`, 'i');
    if (pattern.test(content)) {
      mentions.push(modelId);
    }
  }
  return [...new Set(mentions)];
}

// 计算AI发言意愿分
function calculateWillingness(aiId, userMessage, mentions, debateMode, recentMessages) {
  const persona = AI_PERSONAS[aiId];
  if (!persona) return 0;
  
  let score = 50; // 基础分
  
  // @提及加分 +40
  if (mentions.includes(aiId)) {
    score += 40;
  }
  
  // 辩论模式加分 +10
  if (debateMode) {
    score += 10;
  }
  
  // 新观点加分(检查是否与最近消息重复)
  const recentContents = recentMessages.map(m => m.content).join(' ');
  const similarity = calculateSimilarity(userMessage, recentContents);
  if (similarity < 0.3) {
    score += 10; // 新话题加分
  } else {
    score -= 30; // 重复观点扣分
  }
  
  // 领域匹配加分
  if (persona.preferredTopics.some(topic => 
    userMessage.toLowerCase().includes(topic.toLowerCase())
  )) {
    score += 15;
  }
  
  return Math.min(100, Math.max(0, score));
}
```

```javascript
// 解析AI回复中的引用和社交动作
function parseReplyReference(content, recentMessages) {
  const result = {
    content: content,
    replyTo: null,
    replyQuote: null,
    socialActions: []
  };

  // 解析旧格式引用 【回复消息ID:xxx】
  const idRefMatch = content.match(/【回复消息ID:([^】]+)】/);
  if (idRefMatch) {
    const targetId = idRefMatch[1];
    const targetMsg = recentMessages.find(m => m.id === targetId);
    if (targetMsg) {
      result.replyTo = targetId;
      result.replyQuote = targetMsg.content.substring(0, 50);
      result.content = content.replace(idRefMatch[0], '').trim();
    }
  }

  // 解析新格式引用 > 引用文本
  const quoteRegex = /^>\s*(.+?)$/m;
  const quoteMatch = content.match(quoteRegex);
  if (quoteMatch && !result.replyTo) {
    const quoteText = quoteMatch[1].trim();
    // 通过相似度匹配找到最可能的原消息
    let bestMatch = null;
    let bestScore = 0;
    for (const msg of recentMessages) {
      const sim = calculateSimilarity(quoteText, msg.content);
      if (sim > bestScore && sim > 0.5) {
        bestScore = sim;
        bestMatch = msg;
      }
    }
    if (bestMatch) {
      result.replyTo = bestMatch.id;
      result.replyQuote = quoteText;
      result.content = content.replace(quoteRegex, '').trim();
    }
  }

  // 解析社交动作标记 【点赞】【点踩】【评论:内容】
  const likeMatch = result.content.match(/【点赞】/);
  const dislikeMatch = result.content.match(/【点踩】/);
  const commentMatch = result.content.match(/【评论:(.+?)】/);

  if (likeMatch) {
    result.socialActions.push({ type: 'like' });
    result.content = result.content.replace(/【点赞】/g, '').trim();
  }
  if (dislikeMatch) {
    result.socialActions.push({ type: 'dislike' });
    result.content = result.content.replace(/【点踩】/g, '').trim();
  }
  if (commentMatch) {
    result.socialActions.push({ 
      type: 'comment', 
      content: commentMatch[1].trim() 
    });
    result.content = result.content.replace(/【评论:.+?】/g, '').trim();
  }

  return result;
```

```javascript
async function generateAIResponse(
  aiId, 
  groupId, 
  userMessage, 
  recentMessages, 
  groupMembers, 
  context, 
  isMentioned, 
  userAgents
) {
  const persona = AI_PERSONAS[aiId];
  if (!persona || !persona.responseConfig.enabled) {
    return null;
  }

  const { responseConfig } = persona;
  
  // 计算发言概率
  let probability = responseConfig.responseFrequency;
  if (isMentioned) probability += 0.2; // 被@时增加概率
  if (recentMessages.some(m => m.sender_id === aiId)) {
    probability -= 0.2; // 刚发过言降低概率
  }

  // 随机决定是否发言
  if (Math.random() > probability) {
    return null;
  }

  // 构建提示词
  const { systemPrompt, userMessage: prompt } = buildWeChatStylePrompt(
    persona, userMessage, recentMessages, groupMembers, isMentioned, userAgents
  );

  // 调用AI生成
  let fullResponse;
  try {
    fullResponse = await callAIStream(
      aiId,
      persona,
      prompt,
      [],
      'group_chat',
      groupId,
      groupMembers,
      wsServer,
      context,
      isMentioned,
      userAgents,
      systemPrompt,
      [],
      null,
      null,
      null
    );
  } catch (error) {
    console.error(`[调度引擎] ${aiId} 生成失败:`, error.message);
    return null;
  }

  if (!fullResponse || fullResponse.trim().length === 0) {
    return null;
  }

  // 处理多段消息(用---分隔)
  const segments = fullResponse.split('\n---\n').filter(s => s.trim());
  
  return {
    aiId,
    segments: segments.map(s => s.trim()),
    isAdditional: segments.length > 1,
    originalContent: fullResponse
  };
}
```

```javascript
export function createServer(httpServer) {
  wss = new WebSocketServer({ 
    server: httpServer,
    path: '/ws',
    verifyClient: (info, callback) => {
      // 验证Token
      const token = info.req.headers['sec-websocket-protocol'];
      // 简单验证通过
      callback(true);
    }
  });

  wss.on('connection', (ws, req) => {
    const clientId = uuidv4();
    const userId = extractUserId(req); // 从cookie或query中提取
    
    clients.set(clientId, { ws, userId, rooms: new Set() });
    
    ws.on('message', (data) => {
      try {
        const message = JSON.parse(data.toString());
        handleClientMessage(clientId, message);
      } catch (error) {
        console.error('WebSocket message parse error:', error);
      }
    });
    
    ws.on('close', () => {
      // 清理
      const client = clients.get(clientId);
      if (client) {
        for (const room of client.rooms) {
          const members = groupRooms.get(room);
          if (members) members.delete(clientId);
        }
      }
      clients.delete(clientId);
    });
    
    ws.on('error', (error) => {
      console.error('WebSocket error:', error);
    });
    
    // 发送连接成功
    ws.send(JSON.stringify({
      type: 'connection_status',
      status: 'connected'
    }));
  });

  console.log('WebSocket server created');
  return wss;
}

// 处理客户端消息
function handleClientMessage(clientId, message) {
  const client = clients.get(clientId);
  if (!client) return;
  
  switch (message.type) {
    case 'join_group':
      handleJoinGroup(clientId, message.group_id);
      break;
      
    case 'leave_group':
      handleLeaveGroup(clientId, message.group_id);
      break;
      
    case 'send_message':
      handleSendMessage(clientId, message);
      break;
      
    case 'typing':
      handleTyping(clientId, message);
      break;
      
    default:
      console.warn('Unknown message type:', message.type);
  }
}

// 加入群组房间
function handleJoinGroup(clientId, groupId) {
  const client = clients.get(clientId);
  if (!client) return;
  
  const roomName = `group:${groupId}:${client.userId}`;
  client.rooms.add(roomName);
  
  if (!groupRooms.has(roomName)) {
    groupRooms.set(roomName, new Set());
  }
  groupRooms.get(roomName).add(clientId);
}

// 离开群组房间
function handleLeaveGroup(clientId, groupId) {
  const client = clients.get(clientId);
  if (!client) return;
  
  const roomName = `group:${groupId}:${client.userId}`;
  client.rooms.delete(roomName);
  
  const members = groupRooms.get(roomName);
  if (members) {
    members.delete(clientId);
    if (members.size === 0) groupRooms.delete(roomName);
  }
}

// 发送消息
async function handleSendMessage(clientId, message) {
  const client = clients.get(clientId);
  if (!client) return;
  
  const { group_id, content, file_ids, mentions } = message;
  
  // 创建用户消息
  const userMsg = {
    id: uuidv4(),
    group_id,
    sender_type: 'user',
    sender_id: client.userId,
    sender_name: '用户',
    content,
    content_type: 'text',
    file_ids: file_ids || null,
    mentions: mentions || null,
    likes: 0,
    liked_by: [],
    created_at: new Date().toISOString()
  };
  
  // 存储消息
  await saveUserMessage(userMsg);
  
  // 广播给同群组的其他用户
  const roomName = `group:${group_id}:${client.userId}`;
  broadcastToGroup(group_id, {
    type: 'new_message',
    message: userMsg
  }, clientId); // 排除发送者
  
  // 触发AI调度
  import('../services/scheduler/index.js').then(({ queueAIMessages }) => {
    queueAIMessages(group_id, content, null, { broadcastToGroup });
  });
}

// 广播到群组
export function broadcastToGroup(groupId, data, excludeClientId = null) {
  // 遍历所有用户的房间
  for (const [roomName, memberIds] of groupRooms) {
    if (roomName.startsWith(`group:${groupId}:`)) {
      for (const clientId of memberIds) {
        if (clientId === excludeClientId) continue;
        const client = clients.get(clientId);
        if (client && client.ws.readyState === 1) {
          client.ws.send(JSON.stringify(data));
        }
      }
    }
  }
}

// 处理输入状态
function handleTyping(clientId, message) {
  const client = clients.get(clientId);
  if (!client) return;
  
  broadcastToGroup(message.group_id, {
    type: 'typing',
    userId: client.userId,

  }, clientId);

  }
```

```javascript
  if (braceMatch) {

  }

  const bracketMatch = normalized.match(/\[[\s\S]*\]/);
  if (bracketMatch) {
    try { return JSON.parse(bracketMatch[0]); } catch {}
```

```javascript
import fs from 'fs';
import pdf from 'pdf-parse';
import mammoth from 'mammoth';
import xlsx from 'xlsx';
import { parse as csvParse } from 'csv-parse/sync';

export async function parseFile(filePath, mimeType) {
  const ext = path.extname(filePath).toLowerCase();
  
  switch (mimeType) {
    case 'application/pdf':
      return parsePDF(filePath);
    
    case 'application/msword':
    case 'application/vnd.openxmlformats-officedocument.wordprocessingml.document':
      return parseWord(filePath);
    
    case 'application/vnd.ms-excel':
    case 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet':
      return parseExcel(filePath);
    
    case 'text/csv':
      return parseCSV(filePath);
    
    case 'text/plain':
    case 'text/markdown':
      return parseText(filePath);
    
    default:
      if (['.py', '.js', '.html', '.css', '.json', '.xml', '.ts', '.jsx', '.tsx', '.md'].includes(ext)) {
        return parseCode(filePath);
      }
      return null;
  }
}

async function parsePDF(filePath) {
  const dataBuffer = fs.readFileSync(filePath);
  const data = await pdf(dataBuffer);
  return data.text;
}

async function parseWord(filePath) {
  const result = await mammoth.extractRawText({ path: filePath });
  return result.value;
}

async function parseExcel(filePath) {
  const workbook = xlsx.readFile(filePath);
  const results = [];
  
  for (const sheetName of workbook.SheetNames) {
    const sheet = workbook.Sheets[sheetName];
    const data = xlsx.utils.sheet_to_json(sheet);
    results.push({ sheet: sheetName, data });
  }
  
  return JSON.stringify(results, null, 2);
}

async function parseCSV(filePath) {
  const content = fs.readFileSync(filePath, 'utf-8');
  const records = csvParse(content, { columns: true, skip_empty_lines: true });
  return JSON.stringify(records, null, 2);
}

async function parseText(filePath) {
  return fs.readFileSync(filePath, 'utf-8');
}

async function parseCode(filePath) {
  return fs.readFileSync(filePath, 'utf-8');
}
```

### 11.4 WebSocket客户端完整实现

**文件路径：`frontend/src/services/websocket.ts`**

```typescript
class WebSocketService {
  private ws: WebSocket | null = null;
  private reconnectAttempts = 0;
  private maxReconnectAttempts = 5;
  private reconnectDelay = 1000;
  private listeners: Map<string, Set<Function>> = new Map();
  private isConnected = false;

  connect(token?: string) {
    const protocol = window.location.protocol === 'https:' ? 'wss:' : 'ws:';
    const host = window.location.host;
    const url = `${protocol}//${host}/ws`;
    
    this.ws = new WebSocket(url);
    
    this.ws.onopen = () => {
      console.log('WebSocket connected');
      this.isConnected = true;
      this.reconnectAttempts = 0;
      this.emit('connection_status', { status: 'connected' });
    };

    this.ws.onmessage = (event) => {
      try {
        const data = JSON.parse(event.data);
        this.emit(data.type, data);
      } catch (e) {
        console.error('WebSocket parse error:', e);
      }
    };

    this.ws.onclose = () => {
      this.isConnected = false;
      this.emit('connection_status', { status: 'disconnected' });
      this.reconnect();
    };

    this.ws.onerror = (error) => {
      console.error('WebSocket error:', error);
    };
  }

  private reconnect() {
    if (this.reconnectAttempts >= this.maxReconnectAttempts) {
      this.emit('connection_status', { status: 'failed' });
      return;
    }

    this.reconnectAttempts++;
    this.emit('connection_status', { 
      status: 'reconnecting', 
      attempt: this.reconnectAttempts 
    });

    setTimeout(() => {
      this.connect();
    }, this.reconnectDelay * Math.pow(2, this.reconnectAttempts - 1));
  }

  send(data: any) {
    if (this.ws && this.ws.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify(data));
    }
  }

  joinGroup(groupId: string) {
    this.send({ type: 'join_group', group_id: groupId });
  }

  leaveGroup(groupId: string) {
    this.send({ type: 'leave_group', group_id: groupId });
  }

  sendMessage(groupId: string, content: string, fileIds?: string[], mentions?: string[]) {
    this.send({
      type: 'send_message',
      group_id: groupId,
      content,
      file_ids: fileIds,
      mentions
    });
  }

  on(event: string, callback: Function) {
    if (!this.listeners.has(event)) {
      this.listeners.set(event, new Set());
    }
    this.listeners.get(event)!.add(callback);
  }

  off(event: string, callback: Function) {
    this.listeners.get(event)?.delete(callback);
  }

  private emit(event: string, data: any) {
    this.listeners.get(event)?.forEach(cb => cb(data));
  }

  disconnect() {
    this.ws?.close();
    this.ws = null;
  }
}

export const wsService = new WebSocketService();
```

## 第13章：安全机制

### 13.1 文件安全

| 层级 | 机制 | 实现 |
|------|------|------|
| **第1层** | 扩展名黑名单 | 15种: .exe/.bat/.sh/.ps1/.vbs/.js/.msi/.com/.scr/.dll/.pif/.reg/.wsf/.ws |
| **第2层** | MIME白名单 | 24种允许的类型 |
| **第3层** | Magic Bytes | 读取前8字节校验文件真实性 |
| **第4层** | 大小限制 | 50MB单文件上限 |
| **第5层** | 数量限制 | 最多10个文件 |
| **第6层** | 隔离存储 | 按userId分目录: `uploads/{userId}/` |

### 13.2 输入验证

| 机制 | 实现 | 位置 |
|------|------|------|
| **Zod验证** | validateBody中间件使用Zod Schema验证请求体 | `validators/index.js` |
| **空消息拦截** | 前端trim()空消息不发送 | `MessageInput.tsx` |
| **会话验证** | 基于 `session_token` Cookie 校验会话状态 | `middleware/auth.js` |
| **XSS防护** | 前端使用react-markdown安全渲染 | `MessageContent.tsx` |

### 13.3 数据隔离

| 机制 | 实现 |
|------|------|
| **用户数据库隔离** | 每个用户独立JSON文件: `data/db/{userId}.json` |
| **文件存储隔离** | 每个用户独立目录: `uploads/{userId}/` |
| **WebSocket房间隔离** | 房间名包含userId: `group:{groupId}:{userId}` |
| **写操作互斥锁** | async-mutex按userId加锁 |
